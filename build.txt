# Ставим django.
sudo apt-get install python-pip
pip install django

# Копируем программу из репозитория.
git clone https://github.com/ivan371/webtanks/sources/mysite

# Запускаем сервер
cd webtanks/sources/tanks
python manage.py runserver

# Открываем страничку в браузере
gnome-open http://127.0.0.1:8000/webtanks/

# Чтобы авторизоваться на сервере, необходимо сначала запустить тестовый почтовый сервер в отдельном окне терминала:
python -m smtpd -n -c DebuggingServer localhost:1025

# Затем пойти на страницу решистрации
gnome-open http://127.0.0.1:8000/webtanks/register/

# Ввести данные и зарегаться. После этого на почту придёт уникальный ключ (будет видно сообщение в том окне, где открыт тестовый почтовый сервер)
# Надо скопировать пришедший на почту ключик (вида eb70bb955d108d78b06d71805c38eaf7a6806176) сюда: 
# http://127.0.0.1:8000/accounts/activate/*insert_your_key_here* и перейти по этой ссылке. После этого аккаунт будет активирован. 
# Можно зайти в админку и проверить это. (не забудьте создать суперпользователя!)
gnome-open http://127.0.0.1:8000/admin/

# Можно восстановить пароль, сбросить пароль и тд. — названия файлов говорят сами за себя.
# Теперь наконец можно залогиниться:
gnome-open http://127.0.0.1:8000/webtanks/login/
